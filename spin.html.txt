<script>
    // === Konfigurasi hadiah ===
    // Pastikan 5 file gambar ini sudah diunggah ke folder yang sama dengan file HTML!
    const rewardImages = [
        "x 299 - 3.jpg", // Hadiah 1: Berlian/Gems
        "x 299 - 2.jpg", // Hadiah 2: Gold Badge
        "x 299 - 4.jpg", // Hadiah 3: Crate/Box
        "x 299 - 5.jpg", // Hadiah 4: Hexagonal Gem
        "x 299 - 6.jpg"  // Hadiah 5: Mask/Head
    ];

    // Nama/Deskripsi Hadiah (Sesuaikan dengan gambar)
    const rewardNames = [
        "999 DIAMONDS!",
        "ELITE PASS BADGE!",
        "LEGENDARY CRATE!",
        "RARE GEM FRAGMENT!",
        "MYTHIC MASK SKIN!"
    ];

    const numRewards = rewardImages.length;
    const degreePerSlice = 360 / numRewards;

    const wheel = document.getElementById("wheel");
    const result = document.getElementById("result");
    const spinBtn = document.getElementById("spinBtn");

    // Warna latar belakang untuk setiap slice (opsional, untuk memperindah)
    const colors = ["#b30000", "#cc6600", "#0066cc", "#6600cc", "#00cc66"];

    // Generate slice
    for (let i = 0; i < numRewards; i++) {
        const slice = document.createElement("div");
        slice.className = "slice";
        slice.style.transform = `rotate(${degreePerSlice * i}deg)`;
        slice.style.backgroundColor = colors[i % colors.length]; // Menggunakan warna latar
        
        // Memasukkan gambar dan memutar balik gambar agar tegak lurus
        slice.innerHTML = `<img src="${rewardImages[i]}" style="width:100%;height:100%;transform:rotate(${-degreePerSlice * i}deg);"/>`;
        wheel.appendChild(slice);
    }

    let spinning = false;

    spinBtn.onclick = () => {
        if (spinning) return;
        spinning = true;
        spinBtn.disabled = true;
        result.innerText = "Berputar...";
        
        // Reset rotasi
        wheel.style.transition = 'none';
        wheel.style.transform = 'rotate(0deg)';

        // Delay sebelum spin dimulai
        setTimeout(() => {
            wheel.style.transition = 'transform 4s cubic-bezier(0.1, 0.1, 0.05, 1)';
            
            // Pilih hadiah secara acak (0 sampai 4)
            const winningIndex = Math.floor(Math.random() * numRewards); 
            
            // Hitung sudut yang dibutuhkan agar hadiah berada di bawah indikator
            const stopDegree = 360 - (winningIndex * degreePerSlice) - (degreePerSlice / 2);
            
            // Tambahkan putaran ekstra (misalnya 5 putaran penuh)
            const fullSpins = 5 * 360; 
            const finalRotation = fullSpins + stopDegree;
            
            wheel.style.transform = `rotate(${finalRotation}deg)`;

            setTimeout(() => {
                spinning = false;
                spinBtn.disabled = false;
                result.innerHTML = `ðŸŽ‰ **SELAMAT!** Anda mendapatkan: **${rewardNames[winningIndex]}**!`;
            }, 4200); // Selesai setelah 4.2 detik (sedikit lebih lama dari transisi CSS)
        }, 100);
    };
</script>
